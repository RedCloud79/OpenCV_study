def loadMap():
    with open('level1_map.txt', 'r') as f:

def checkMovable(dy, dx) -> bool:

    if (
            (0 <= dy and dy <= 8) and (dx >= 0 and dx <= 8) # 지정된 범위
            and (mapdata[dy][dx] != 'x') # 지정된 타일 확인
        ):
            return True
    
    return False
